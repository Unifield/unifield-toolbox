#for i in {2..18}; do cp 1.xml $i.xml; done

for account_line in {2..18}; do

	# first value is 5
	file_nb=$((($account_line-1)*4+1))

	cp 1.xml $file_nb.xml;

	file_two=$(($file_nb+1))
	cp 2.xml $file_two.xml;

	file_three=$(($file_nb+2))
	cp 3.xml $file_three.xml;

	file_four=$(($file_nb+3))
	cp 4.xml $file_four.xml;


# account_move in 1, 2, 4
# account_move_line in 2,3


	sed -i "s/account_move\/1/account_move\/$account_line/" $file_nb.xml;
	sed -i "s/account_move\/1/account_move\/$account_line/" $file_two.xml;
	sed -i "s/account_move\/1/account_move\/$account_line/" $file_four.xml;


	sed -i "s/account_move_line\/1/account_move_line\/$file_nb/" $file_two.xml;
	sed -i "s/account_move_line\/2/account_move_line\/$file_two/" $file_two.xml;
	sed -i "s/account_move_line\/1/account_move_line\/$file_nb/" $file_three.xml;
	sed -i "s/account_move_line\/2/account_move_line\/$file_two/" $file_three.xml;		
done
